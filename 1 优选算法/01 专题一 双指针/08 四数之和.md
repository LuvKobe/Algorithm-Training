### 1. 题目链接

[18. 四数之和](https://leetcode.cn/problems/4sum/)


### 2. 解法（排序 + 双指针）

**算法思路：**
- 依次固定一个数 `a`;
- 在这个数 `a` 的后面区间上，利用「三数之和」找到三个数。使得这三个数的和等于 `target - a` 即可。

### 3. 算法代码

**C++实现:**
```cpp
class Solution {
    typedef long long LL;   // int会溢出, 所以换成long long
public:
    vector<vector<int>> fourSum(vector<int>& nums, int target) {
        // 1. 定义返回容器
        vector<vector<int>> ret;

        // 2. 排序
        sort(nums.begin(), nums.end());

        // 3. 利用双指针解决问题
        int n = nums.size();
        for (int i = 0; i < n; )    // 固定第一个数a
        {
            // 利用三数之和
            for (int j = i + 1; j < n; )    // 固定第二个数b
            {
                int left  = j + 1, right = n - 1;
                LL aim = (LL)target - nums[i] - nums[j];  // int会溢出
                while (left < right)    // 找一个等于aim的数
                {
                    int sum = nums[left] + nums[right];
                    if (sum > aim)
                    {
                        right--;
                    }
                    else if (sum < aim)
                    {
                        left++;
                    }
                    else 
                    {
                        ret.push_back({nums[i], nums[j], nums[left], nums[right]});
                        left++, right--;

                        // 对left去重
                        while (left < right && nums[left] == nums[left - 1])
                            left++;
                        
                        // 对right去重
                        while (left < right && nums[right] == nums[right + 1])
                            right--;
                    }
                }

                // 对j去重
                j++;
                while (j < n && nums[j] == nums[j - 1])
                    j++;
            }

            // 对i去重
            i++;
            while (i < n && nums[i] == nums[i - 1])
                i++;
        }
        return ret;
    }
};
```
